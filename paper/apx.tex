\section{Graphoid Axioms}\label{apx:graphoid}
\begin{definition}[Graphoid Axioms]\label{def:graphoid}
  For disjoint subsets $\rmX,\rmY,\rmZ,\rmW\subseteq\verts$, a
  semi-graphoid independence model $\indeps{\cdot}$ obeys the following
  properties~\citep{dawid1979,pearl1987}:
  \begin{enumerate}
  \item Symmetry:
    $\rmX \indep{} \rmY \mid \rmZ \implies \rmY\indep{}\rmX\mid\rmZ$
  \item Decomposition:
    $\rmX \indep{} \rmY \cup \rmW \mid \rmZ \implies
    \rmX\indep{}\rmY\mid\rmZ$
  \item Weak Union:
    $\rmX\indep{}\rmY\cup\rmW\mid\rmZ \implies \rmX\indep{}\rmY \mid
    \rmW\cup\rmZ$
  \item Contraction:
    $(\rmX\indep{}\rmY \mid \rmW\cup\rmZ) \wedge
    (\rmX\indep{}\rmW\mid\rmZ)\implies \rmX\indep{}\rmY\cup\rmW\mid\rmZ$
  \end{enumerate}
  The independence model is called a graphoid if it also obeys the
  following additional property~\citep{pearl1985}:
  \begin{enumerate}[resume]
  \item Intersection:
    $(\rmX \indep{} \rmY \mid \rmZ \cup \rmW) \wedge
    (\rmX \indep{} \rmW \mid \rmZ \cup \rmY)
    \implies \rmX \indep{} \rmY \cup \rmW \mid
    \rmZ$
  \end{enumerate}
  All graph independence models $\indeps{\gr}$ are guaranteed to
  obey all the graphoid axioms~\citep{lauritzen2018}. However,
  probabilistic independence models $\indeps{\pr}$ are only guaranteed to
  obey the semi-graphoid axioms. However if the joint distribution is
  strictly positive, then $\indeps{\pr}$ is guaranteed to be a
  graphoid as well~\citep{pearl1988,pearl1989}.
\end{definition}

%%%% k dep to assoc
\kdeptoassoc*
\begin{proof}[Proof by induction]
  When $\rmZ'$ is an empty set, and therefore a subset of $\rmZ$ with
  cardinality $0$, we know by definition that $X$ and $Y$ being
  $0$-order dependent is equivalent to them being $0$-order associated,
  \begin{gather}
    \label{eq:12}
    \begin{aligned}
      &\kdep{\verts}{X,Y}{\emptyset}\\
      &=\kassoc{\rmS}{Y,X}{\emptyset} \wedge
      \forall \rmZ^{*} \subset \emptyset
        : \neg\,\kassoc{\rmS}{Y,X}{\rmZ^{*}}\\
      &=\kassoc{\rmS}{Y,X}{\emptyset},
    \end{aligned}
  \end{gather}
  since there are no proper subsets of the empty set to begin with.
  
  Now assume that $Y$ and $X$ are not $m$-order associated---where
  $m \leq k < |\rmZ|$---w.r.t. any subsets in $\powerleq{k}{\rmZ}$,
  \begin{equation}
    \label{eq:k-dep-to-assoc-eq1}
    \forall \rmZ^{*} \in \powerleq{k}{\rmZ} :
    \neg\,\kassoc{\rmS}{Y,X}{\rmZ^{*}}.
  \end{equation}
  Then we know that if $Y$ and $X$ are $(k+1)$-order associated with any
  subset $\rmZ' \in \powereq{k+1}{\rmZ}$,
  \begin{equation}
    \label{eq:k-dep-to-assoc-eq2}
    \kassoc{\rmS}{Y,X}{\rmZ'},
  \end{equation}
  taking the conjunction of \Cref{eq:k-dep-to-assoc-eq1} and
  \Cref{eq:k-dep-to-assoc-eq2} results in $Y$ and $X$ being
  $(k+1)$-order dependent w.r.t. $\rmZ'$,
  \begin{align}
    &\kassoc{\rmS}{Y,X}{\rmZ'} \wedge
      \forall \rmZ^{*} \in \powerleq{k}{\rmZ'} :
      \neg\,\kassoc{\rmS}{Y,X}{\rmZ^{*}}\nonumber\\
    &\iff \kdep{\rmS}{Y,X}{\rmZ'}, \label{eq:14}
  \end{align}
  since $\powerleq{k}{\rmZ'}\subseteq\powerleq{k}{\rmZ}$ and we know
  that---at the worst---$Y$ and $X$ are $(k+1)$-order associated when
  $k=|\rmZ|-1$.
\end{proof}

%%%% k dep to inter
\kdepinter*
\begin{proof}
  First assume that $\kdep{\rmS}{Y,X}{\rmZ}$ which from its definition
  in \Cref{def:k-dep} implies that $Y$ and $X$ are not associated given
  any proper subset of $\rmZ$,
  \begin{gather}
    \begin{aligned}
      &\kdep{\rmS}{Y,X}{\rmZ}\\
      &\implies
      \forall \rmZ' \subset \rmZ \:\: \exists \rmS' \subseteq \rmS :
      Y \indep{\pr} X \mid \rmZ' \cup \rmS'
    \end{aligned}
    .\label{eq:k-dep-inter-eq2}
  \end{gather}
  Furthermore, assume that for some $Z\in\rmZ$, $Y$ and $Z$ are
  not associated w.r.t. $X\cup\rmZ\setminus Z$,
  \begin{equation}
    \label{eq:k-dep-inter-eq1}
    \exists Z \in \rmZ \:\: \exists \rmS' \subseteq \rmS:
    Y \indep{\pr} Z \mid X \cup \rmS' \cup \rmZ \setminus Z.
  \end{equation}
  Note that the statement in \Cref{eq:k-dep-inter-eq1} is essentially
  the opposite of what we wish to prove.
  
  By the contraction and weak union graphoid axioms in
  \Cref{def:graphoid} we know that for some $\rmS'\subseteq\rmS$,
  \begin{equation}
    \begin{aligned}
      &(Y \indep{\pr} Z \mid X \cup \rmS' \cup \rmZ \setminus Z)
        \wedge (Y \indep{\pr} X \mid \rmS' \cup \rmZ \setminus Z)\\
      &\implies Y \indep{\pr} \{X, Z\} \mid \rmS' \cup \rmZ \setminus Z\\
      &\implies Y \indep{\pr} X \mid \rmS' \cup \rmZ
    \end{aligned}\label{eq:k-dep-inter-eq4}
  \end{equation}
  However, referring back to the definition of a $k$-order dependance in
  \Cref{def:k-dep}, we known that our initial assumption of
  $\kdep{\rmS}{Y,X}{\rmZ}$ implies that,
  \begin{gather}
    \label{eq:k-dep-inter-eq3}
    \kdep{\rmS}{Y,X}{\rmZ} \implies
    \forall \rmS' \subseteq \rmS : Y \dep{\pr} X \mid \rmS' \cup \rmZ,
  \end{gather}
  which causes a contradiction with the conclusion in
  \Cref{eq:k-dep-inter-eq4}. Therefore, the conjunction of our initial
  assumptions must be false. Taking the negation of this conjunction
  leads to,
  \begin{align*}
    &\neg \bigl(
    \kdep{\rmS}{Y,X}{\rmZ} \wedge\\
    &\qquad\exists Z \in \rmZ \:\: \exists \rmS' \subseteq \rmS: 
    Y \indep{\pr} Z \mid X \cup \rmS \cup \rmZ \setminus Z
    \bigr)\\
    &=\neg\kdep{\rmS}{Y,X}{\rmZ} \vee\\
    &\qquad\forall Z \in \rmZ \:\: \forall \rmS' \subseteq \rmS: 
    Y \dep{\pr} Z \mid X \cup \rmS \cup \rmZ \setminus Z\\
    &=\kdep{\rmS}{Y,X}{\rmZ} \implies
    \forall Z \in \rmZ :
      \kassoc{\rmS}{Y,Z}{X\cup\rmZ\setminus Z}.
  \end{align*}
  Which in addition to the implication we know to be true in
  \Cref{eq:k-dep-inter-eq3} results in,
  \begin{equation}
    \label{eq:29}
    \kdep{\rmS}{Y,X}{\rmZ} \implies \kinter{\rmS}{Y, \rmZ \cup X},
  \end{equation}
  concluding our proof.
\end{proof}


%%%% Dep subset
\depsubset*
\begin{proof}
  Recall that $\kdep{\rmS}{Y,X}{\rmZ}$ is defined in \Cref{def:k-dep} as
  \begin{equation}
    \begin{aligned}
      &\kdep{\rmS}{Y,X}{\rmZ}\\
      &:= \kassoc{\rmS}{Y,X}{\rmZ} \wedge
        \forall \rmZ' \subset \rmZ : \neg\,\kassoc{\rmS}{Y,X}{\rmZ'}.
    \end{aligned}\tag{\ref{eq:k-dep-dep}}
  \end{equation}
  Therefore, to prove \Cref{eq:subset-res}, we will first explore how
  reducing the separating set to $\rmS'\subseteq\rmS$ affects the
  first term of the conjunction in \Cref{eq:k-dep-dep},
  \begin{align}
    \kdep{\rmS}{Y,X}{\rmZ}
    \implies &\kassoc{\rmS}{Y,X}{\rmZ}\\
    \iff &\forall \rmS^{*} \subseteq \rmS :
           Y \dep{\pr} X \mid \rmZ \cup \rmS^{*}\label{eq:subset-0}\\
    \implies
         &\forall \rmS^{*} \subseteq \rmS' :
           Y \dep{\pr} X \mid \rmZ \cup \rmS^{*}\label{eq:subset-1}\\
    \iff &\kassoc{\rmS'}{Y,X}{\rmZ}\label{eq:subset-2}.
  \end{align}
  The implication in \Cref{eq:subset-1} follows from the
  fact that $\mathcal{P}(\rmS')\subseteq\mathcal{P}(\rmS)$ and therefore
  the set of conditional dependencies in \Cref{eq:subset-1} is a subset
  of the conditional dependencies in \Cref{eq:subset-0}.

  Furthermore, using the result from \Cref{lem:k-dep-inter}, we can show
  that,
  \begin{align}
    &\kdep{\rmS}{Y,X}{\rmZ}\implies \kinter{\rmS}{Y,X\cup\rmZ}\\
    &\implies \forall Z \in\rmZ :
               \kassoc{\rmS}{Y,Z}{X\cup\rmZ\setminus Z}\\
    &\implies \forall Z \in\rmZ :
      \kassoc{\rmS'}{Y,Z}{X\cup\rmZ\setminus Z},
      \label{eq:subset-3}
  \end{align}
  where the implication in \Cref{eq:subset-3} follows from the previous
  implication between \Cref{eq:subset-0,eq:subset-2}.
  Therefore, combining the implications in
  \Cref{eq:subset-2,eq:subset-3} we obtain,
  \begin{align}
    \kdep{\rmS}{Y,X}{\rmZ}
    \implies\kinter{\rmS'}{Y,X\cup\rmZ},
  \end{align}
  which concludes this proof.
\end{proof}

\citest*

\begin{proof}
  The function \textit{find\_cond} first iterates through all the
  subsets $\rmX'\in\powerleq{k-|\rmX|}{\rmS}$ and for each $\rmX'$ where
  $Y \dep{\pre} X \mid \rmX'$ (line 9 in \textit{find\_cond}), it passes
  $\rmX'$ to the function \textit{no\_seps} (line 10 in
  \textit{find\_cond}) to determine if there are any subsets
  $\rmS'\in\powerleq{l}{\rmS\setminus\rmX'}$ that results in
  $Y \dep{\pre} X \mid \rmX'\cup \rmS'$ (line 3 in \textit{no\_seps}).
  As soon as \textit{no\_seps} finds such a $\rmS'$, it returns
  \texttt{False} (line 4 in \textit{no\_seps}), and \textit{find\_cond}
  will need to go to the next iteration. Otherwise, if \textit{no\_seps}
  is unable to find any such subset $\rmS'$, it returns \texttt{True}
  (line 5 in \textit{no\_seps}), and \textit{find\_cond} then
  immediately returns $\rmZ=\rmX'\cup\rmX$ (line 11 in
  \textit{find\_cond}) as there are no $l$-bounded subsets of $\rmS$
  that renders $Y$ and $X$ conditionally independent given
  $\rmZ$. However, if \textit{find\_cond} completes all its iteration
  without returning early, then we know that all subset $\rmX'$ has some
  $l$-bounded subset $\rmS'$ that will cause
  $Y\indep{\pre}X\mid\rmX'\cup\rmX\cup\rmS'$. Therefore
  \textit{find\_cond} just returns the emptyset in this case (line 12 in
  \textit{find\_cond}).
\end{proof}

\assocmine*
\begin{proof}
  During each function call, \textit{find\_inter\_dep} is passed
  some--potentially empty---set $\rmX$ and it then iterates through each
  varible $X$ that has not been included in $\rmX$ yet,
  $X\in\verts\setminus(Y \cup \rmX \cup \rmS)$. In each iteration it
  calls the function \textit{find\_cond} to find a subset
  $\rmX' \in \powerleq{k-|\rmX|}{\rmS}$ such that
  \begin{equation*}
    \forall \rmS' \in \powerleq{l}{\rmS \setminus \rmX'} :
    Y\dep{\pre}X  \mid \rmX \cup \rmX' \cup \rmS'.
  \end{equation*}
  When \textit{find\_cond} fails to find such a set $\rmX'$, it just
  returns an empty set, causing \textit{find\_inter\_dep} to check the
  next variable $X$ not already in $\rmX\cup\rmS$. 
  
  Otherwise, when a suitable set $\rmX'$ is found, \textit{find\_cond}
  returns $\rmZ=\rmX'\cup\rmX$ and $\rmZ$ is then passed on to the
  function \textit{make\_strict} to remove any superfluous variables
  resulting in the set $\rmZ'\subseteq(\rmZ\cup X)$ such that,
  \begin{align*}
    &\forall Z \in \rmZ'\:\:
    \forall \rmS' \in \powerleq{l}{\rmS}:
    Y \dep{\pre} Z \mid \rmZ \cup \rmS' \setminus Z\\
    &\iff \forall Z \in \rmZ' :
      \lassoc{\rmS}{Y,Z}{\rmZ\setminus Z}\\
    &\iff \linter{\rmS}{Y,X\cup\rmX\cup\rmX'}.
  \end{align*}
  Furthermore, since \textit{find\_inter\_dep} explores the enumeration
  tree of the set $\verts \setminus (Y \cup \rmS)$ up to subsets of
  cardinality $k$ (lines 13 to 17 in \textit{find\_inter\_dep}) , we
  know that every subset
  $\rmX\in\powerleq{k}{\verts \setminus (Y \cup \rmS)}$ will be
  explored---barring any early returns. In combination with
  \textit{find\_cond} exploring the subsets
  $\rmX' \in \powerleq{k-|\rmX|}{\rmS}$, \textit{find\_inter\_dep}
  returns $X\cup\rmZ$ where $\rmZ=\rmX\cup\rmX'$ if and only if
  \begin{align*}
    \exists \rmZ \in \powerleq{k}{\hat{\verts}} \:\:
    \exists X \in \hat{\verts}\setminus(\rmX\cup\rmS) :
    \linter{\rmS}{Y,X\cup\rmZ}.
  \end{align*}
  where $\hat{\verts}=\verts\setminus Y$. If no such $X$ and $\rmZ$ can
  be found, \textit{find\_inter\_dep} will return an empty set when
  every subset $\rmX\in\powerleq{k}{\verts\setminus (Y\cup\rmS)}$ has
  been explored.
\end{proof}

\algkomb*
\begin{proof}
  First let $\hat{\verts}=\verts\setminus Y$.  At each iteration of the
  grow phase, if there is a variable $X$ and subset $\rmZ$ where
  following logical statement is true,
  \begin{equation*}
    \exists \rmZ \in \powerleq{k}{\hat{\verts}} \:\:
    \exists X \in \hat{\verts}\setminus(\rmX\cup\rmS) :
    \linter{\rmS}{Y,X\cup\rmZ},
  \end{equation*}
  then the function \textit{find\_inter\_dep} will return the set
  $(X\cup\rmZ)$. Furthermore, we know that under the $k$-order
  faithfulness assumption, for some
  existing---potentially empty---candidate Markov blanket 
  $\rmS\subseteq \subverts$, $k'\leq k$, and for all possible $X$,
  \begin{align}
    &Y \dep{\gr} X \mid \verts \iff X \in \mbs{\gr}(Y)
      \label{eq:komb-1}\\
    &\implies \exists \rmZ \in \powereq{k'}{\subverts}:
      \kdep{\verts}{Y,X}{\rmZ}
      \label{eq:komb-2}\\
    &\implies \exists \rmZ \in \powereq{k'}{\subverts}:
      \kinter{\rmS}{Y,X\cup\rmZ}\label{eq:komb-3}\\
    &\implies \exists \rmZ \in \powereq{k'}{\subverts}:
      \linter{\rmS}{Y,X\cup\rmZ}\label{eq:komb-4},
  \end{align}
  where \Cref{eq:komb-1,eq:komb-2} is true by definition, the
  implication in \Cref{eq:komb-3} follows from the result in
  \Cref{lem:subset}, and \Cref{eq:komb-4} is true under the $l$-bounded
  separator assumption in \Cref{def:bound-sep}.
  
  Therefore, we know that whenever there exists some $X$ that is in the
  true Markov blanket of $Y$ but not yet included in the candidate
  Markov blanket $\rmS$, thanks to the implication in \Cref{eq:komb-4},
  we know that \textit{find\_inter\_dep} will be able to find and
  approapiate $\rmZ$ such that $\linter{\rmS}{Y,X\cup\rmZ}$ is true and
  the set $X\cup\rmZ$ will be added into $\rmS$.

  Therefore once \textit{find\_inter\_dep} starts to return empty sets,
  we know that there is no more varibles left in
  $\hat{\verts}\setminus\rmS$ that are in the true Markov blanket of
  $Y$,
  \begin{equation*}
    \mbs{\gr}(Y) \subseteq \rmS.
  \end{equation*}

  During the shrink phase, {\method} iteratively calls
  \textit{find\_cond} on every variable in $\rmS$ and removes that
  variable from $\rmS$ if \textit{find\_cond} returns an empty set. We
  know that given the variable $X$ and initial conditioning set
  $\emptyset$, \textit{find\_cond} returns the empty set if,
  \begin{align*}
    &\forall \rmZ\in\powerleq{k}{\rmS\setminus X}\:\:
    \exists \rmS'\in\powerleq{l}{\rmS\setminus(X\cup\rmZ)} :\\
    &Y \indep{\pre} X \mid \rmZ \cup \rmS'\\
    &\iff \\
    &\forall \rmZ\in\powerleq{k}{\subverts}\:\:
    \exists \rmS'\in\powerleq{l}{\rmS\setminus X} :\\
    &Y \indep{\pre} X \mid \rmZ \cup \rmS' \\
    &\iff Y \indep{\gr} X \mid (\rmS\setminus X)
  \end{align*}
  where the biconditionals follow from the global Markov property since
  $\mbs{\gr}(Y)\subseteq\rmS$ and the $l$-bounded separator
  assumption.

  Therefore, whenever a variable $X\in\rmS$ is not in the Markov blanket
  of $Y$, \textit{find\_cond} returns an empty set and $X$ is removed
  from $\rmS$. As such, after the shrink phase, the subset $\rmS$ is
  exactly the graphical Markov blanket of $Y$, $\rmS=\mbs{\gr}(Y)$.
\end{proof}

\kombcomplexity*
\begin{proof}
  We charge $O(N)$ to each \gls{ci} test and bound the running time by
  analysing the three sub-algorithms of \Cref{sec:algo} from the
  innermost outwards.

  First consider \textit{find\_cond} (\Cref{alg:ci-test}) for a fixed
  candidate variable $X$ and external dependence set $\rmX$ with
  $|\rmX|=i$. It iterates over every subset
  $\rmX'\in\powerleq{k-i}{\rmS}$, of which there are at most
  $\sum_{i'=0}^{k-i}\binom{K}{i'}$ since $|\rmS|\leq K$. For each $\rmX'$
  it performs one \gls{ci} test and, whenever that test indicates a
  dependence, calls \textit{no\_seps}, which iterates over every
  separator $\rmS'\in\powerleq{l}{\rmS\setminus\rmX'}$ and performs one
  further \gls{ci} test per separator. Writing $|\rmX'|=i'$, there are at
  most $\sum_{j=0}^{l}\binom{K-i'-i}{j}$ such separators, so a single
  call to \textit{find\_cond} costs
  \begin{equation}
    \label{eq:cost-findcond}
    O\!\left(
      \sum_{i'=0}^{k-i}\binom{K}{i'}
      \sum_{j=0}^{l}\binom{K-i'-i}{j}\, N
    \right),
  \end{equation}
  where the convention $\binom{n}{m}=0$ for $n<m$ accounts for the
  separator set $\rmS\setminus\rmX'$ being exhausted.

  Next consider \textit{find\_inter\_dep} (\Cref{alg:assoc-mine}). It
  explores the enumeration tree of $\verts\setminus Y$ up to subsets of
  cardinality $k$ (lines 13--17 of \textit{find\_inter\_dep}), and hence
  visits each of the $\sum_{i=0}^{k}\binom{n}{i}$ dependence sets
  $\rmX\in\powerleq{k}{\verts\setminus Y}$. For every such $\rmX$ it
  loops over the $O(n)$ candidate variables
  $X\in\verts\setminus(Y\cup\rmS\cup\rmX)$ and invokes
  \textit{find\_cond}. Multiplying these two factors by
  \Cref{eq:cost-findcond} bounds the cost of a single association-mining
  sweep by
  \begin{equation}
    \label{eq:cost-sweep}
    O\!\left(
      n \sum_{i=0}^{k}\binom{n}{i}
      \left[
        \sum_{i'=0}^{k-i}\binom{K}{i'}
        \sum_{j=0}^{l}\binom{K-i'-i}{j}\, N
      \right]
    \right).
  \end{equation}

  Finally, consider {\method} itself (\Cref{alg:komb}). Each successful
  call to \textit{find\_inter\_dep} in the grow phase returns a
  non-empty set $X\cup\rmZ$ that is added to $\rmS$, so the candidate
  Markov blanket grows by at least one variable per successful call; as
  $|\rmS|\leq K$ throughout, the grow phase performs at most $O(K)$
  association-mining sweeps. The shrink phase calls \textit{find\_cond}
  for each of the at most $K$ variables in $\rmS$, the cost of which is
  dominated by that of the grow phase. Multiplying \Cref{eq:cost-sweep}
  by these $O(K)$ sweeps yields the bound in \Cref{eq:complexity},
  completing the proof.
\end{proof}

\section{Ablation}\label{apx:ablation}
For completeness, we report the full results of the experiments in
\Cref{sec:exp-real} for every method and every $(k,l)$ configuration of
{\method}. \Cref{tab:real-full} gives the performance of all methods,
while \Cref{tab:runtime-full} gives their runtimes in seconds. In both
tables, ``--'' denotes a configuration that failed to complete within the
allotted time budget.

\Cref{tab:kl-ablation} summarises the F1 score of {\method} across all
synthetic and benchmark datasets for every $(k,l)$ configuration. F1
generally improves from $k=0$ to $k=2$---higher orders expose
dependencies that lower orders miss---but the gains flatten or reverse
beyond $k=2$ as the larger conditioning sets incur both higher runtime
(\Cref{tab:runtime-full}) and more error-prone \gls{ci} tests. Setting
$l=k$ with $k\leq 2$ therefore offers a robust default, the heuristic we
